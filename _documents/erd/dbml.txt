Table AdminAction {
  admin_action_id integer [primary key]
  review_id integer
  admin_id integer
  action_type_id integer
  action_date timestamp
}

Table ActionType {
  action_type_id integer [primary key]
  action_type_name varchar [unique]
}

Table Course {
  course_id integer [primary key]
  course_code varchar
  course_detail_id integer
  course_term_id varchar
  course_section varchar
  course_delivery_format varchar
}

Table CourseTerm {
  course_term_id integer [primary key]
  course_season varchar
  course_year integer
}

Table CourseDetail {
  course_detail_id integer [primary key]
  course_name varchar
  course_description varchar
}

Table Policy {
  policy_id integer [primary key]
  policy_name varchar
  policy_description varchar
}

Table Professor {
  professor_id integer [primary key]
  first_name varchar
  last_name varchar
}

Table ProfessorCourse {
  professor_course_id integer [primary key]
  professor_id integer
  course_id integer
}

Table Question {
  question_id integer [primary key]
  question_text varchar
  review_type_id integer
}

Table Review {
  review_id integer [primary key]
  review_type_id integer
  review_status_id integer
  professor_course_id integer
  user_id integer
}

Table ReviewAnswer {
  review_answer_id integer [primary key]
  review_question_id integer
  answer varchar
}

Table ReviewHistory {
  review_history_id integer [primary key]
  review_id integer
  review_status_id integer
  changed_by integer
}

Table ReviewPolicy {
  review_policy integer [primary key]
  review_id integer
  policy_id integer
  policy_violated boolean
}

Table ReviewQuestion {
  review_question_id integer [primary key]
  review_id integer
  question_id integer
}

Table ReviewStatus {
  review_status_id integer [primary key]
  status_name varchar
}

Table ReviewType {
  review_type_id integer [primary key]
  review_type_name varchar
}

Table User {
  user_id integer [primary key]
  full_name varchar
  email varchar [unique]
  role_id integer
}

Table UserRole {
  role_id integer [primary key]
  role_name varchar
}

Table UserProfile {
  profile_id integer [primary key]
  user_id integer
  bio text
}
